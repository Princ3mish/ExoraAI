# Phase 6.1: Backend Foundation — AIEvent, logEvent, Email Service, Events Endpoint

## Objective
Establish the three core backend pillars for Phase 6:
1. `AIEvent` Prisma model — persistent event log backbone
2. `logEvent` shared utility — used by all services to write structured events
3. `email.service.js` — Nodemailer Gmail SMTP (fire-and-forget)
4. `/api/events` polling endpoint — delivers events to frontend

- `backend/src/services/email.service.js` exports `sendEmail` and `sendBulkEmail`
- `GET /api/events` returns 200 with `{ events: [], total: 0 }` when authenticated
- `backend/src/app.js` mounts `/api/events` route
- `backend/package.json` contains `nodemailer` dependency

  <action>
    Create `backend/src/utils/logEvent.js`:

    ```js
    import { PrismaClient } from '@prisma/client';
    import logger from './logger.js';

    const prisma = new PrismaClient();

    /**
     * Write a structured event to AIEvent table AND emit a Winston log.
     * Non-throwing: DB write failure is logged and swallowed.
     *
     * @param {object} opts
     * @param {string} opts.type       - AI_ACTION | SYSTEM | USER_RESPONSE | ERROR | SIMULATION
     * @param {string} opts.message    - Human-readable description
     * @param {string} [opts.status]   - 'success' | 'pending' | 'failed'
     * @param {string} [opts.meetingId]
     * @param {string} [opts.userId]
     * @param {object} [opts.metadata] - Extra serializable data
     */
    export async function logEvent({ type, message, status = 'success', meetingId, userId, metadata }) {
      const winstonMeta = { type, status, meetingId, userId, ...metadata };
      if (status === 'failed') {
        logger.error(`[AIEvent] ${message}`, winstonMeta);
      } else {
        logger.info(`[AIEvent] ${message}`, winstonMeta);
      }

      try {
        await prisma.aIEvent.create({
          data: {
            type,
            message,
            status,
            meetingId: meetingId || null,
            userId: userId || null,
            metadata: metadata || null,
          },
        });
      } catch (err) {
        logger.error('[logEvent] Failed to persist AIEvent to DB', { error: err.message, type, message });
      }
    }
    ```

  <action>
    1. Run inside `backend/`: `npm install nodemailer`

    2. Create `backend/src/services/email.service.js`:

    ```js
    import nodemailer from 'nodemailer';
    import logger from '../utils/logger.js';
    import { logEvent } from '../utils/logEvent.js';

    const createTransporter = () => {
      if (!process.env.EMAIL_USER || !process.env.EMAIL_PASS) {
        logger.warn('[EmailService] EMAIL_USER or EMAIL_PASS not configured');
        return null;
      }
      return nodemailer.createTransport({
        service: 'gmail',
        auth: { user: process.env.EMAIL_USER, pass: process.env.EMAIL_PASS },
      });
    };

    export async function sendEmail({ to, subject, body, meetingId, userId }) {
      const transporter = createTransporter();
      if (!transporter) {
        await logEvent({ type: 'ERROR', message: `Email skipped for ${to}: credentials not configured`, status: 'failed', meetingId, userId, metadata: { to } });
        return;
      }
      try {
        await transporter.sendMail({ from: `"Exora AI" <${process.env.EMAIL_USER}>`, to, subject, text: body });
        await logEvent({ type: 'AI_ACTION', message: `Email sent to ${to}`, status: 'success', meetingId, userId, metadata: { to, subject } });
        logger.info(`[EmailService] Sent to ${to}`, { to, meetingId });
      } catch (err) {
        await logEvent({ type: 'ERROR', message: `Email failed for ${to}: ${err.message}`, status: 'failed', meetingId, userId, metadata: { to, error: err.message } });
        logger.error(`[EmailService] Failed to send to ${to}`, { error: err.message });
        // Never rethrow — caller must not crash on email failure
      }
    }

    export async function sendBulkEmail({ recipients, subject, body, meetingId }) {
      await Promise.allSettled(
        recipients.map(({ email, userId }) => sendEmail({ to: email, subject, body, meetingId, userId }))
      );
    }
    ```

    ```js
    import { PrismaClient } from '@prisma/client';
    import logger from '../../utils/logger.js';

    const prisma = new PrismaClient();

    /**
     * GET /api/events
     * Returns AIEvent records for real-time frontend polling.
     * Query: ?since=ISO_timestamp, ?limit=number
     */
    export const getEvents = async (req, res, next) => {
      try {
        const { since, limit } = req.query;
        const take = Math.min(parseInt(limit) || 50, 100);
        const where = {};
        if (since) {
          const sinceDate = new Date(since);
          if (!isNaN(sinceDate.getTime())) {
            where.createdAt = { gt: sinceDate };
          }
        }
        const events = await prisma.aIEvent.findMany({
          where,
          orderBy: { createdAt: 'desc' },
          take,
        });
        const chronological = events.reverse();
        res.json({ success: true, data: { events: chronological, total: chronological.length, timestamp: new Date().toISOString() } });
      } catch (err) {
        logger.error('[EventsController] Failed to fetch events', { error: err.message });
        next(err);
      }
    };
    ```
