    ```js
    /**
     * Phase 6: Fire-and-forget email sending for new meeting invitations.
     * Fetches participant email addresses and sends AI-drafted invite emails.
     * Non-blocking: errors are caught and logged by the email service itself.
     */
    const _sendMeetingInviteEmails = async (meeting, organizerId) => {
      try {
        // Fetch full participant user records to get emails
        const participants = await prisma.participant.findMany({
          where: { meetingId: meeting.id },
          include: { user: { select: { id: true, name: true, email: true } } },
        });

        await logEvent({
          type: 'AI_ACTION',
          message: `AI drafting invite email for "${meeting.title}"`,
          status: 'pending',
          meetingId: meeting.id,
          userId: organizerId,
        });

        // Generate AI email draft
        let emailSubject = `Meeting Invitation: ${meeting.title}`;
        let emailBody = `You have been invited to "${meeting.title}".\n\nStart: ${meeting.startTime.toISOString()}\nEnd: ${meeting.endTime.toISOString()}\n\nPlease log in to Exora AI to respond.`;

        try {
          const aiResult = await generateCompletion({
            taskType: 'DRAFT_EMAIL',
            input: { meetingId: meeting.id, contextType: 'invite' },
            metadata: { userId: organizerId, meetingId: meeting.id },
          });
          if (aiResult?.subject) emailSubject = aiResult.subject;
          if (aiResult?.body) emailBody = aiResult.body;
          await logEvent({
            type: 'AI_ACTION',
            message: `AI email draft ready for "${meeting.title}"`,
            status: 'success',
            meetingId: meeting.id,
            userId: organizerId,
          });
        } catch (aiErr) {
          await logEvent({
            type: 'ERROR',
            message: `AI email draft failed, using fallback template: ${aiErr.message}`,
            status: 'failed',
            meetingId: meeting.id,
            userId: organizerId,
          });
        }

        // Send to all participants
        const recipients = participants.map(p => ({ email: p.user.email, userId: p.user.id }));
        await sendBulkEmail({ recipients, subject: emailSubject, body: emailBody, meetingId: meeting.id });
      } catch (err) {
        logger.error('[_sendMeetingInviteEmails] Unexpected error', { error: err.message, meetingId: meeting.id });
      }
    };
    ```

    2. In `backend/src/api/ai/ai.controller.js`, add a new export at the bottom:
    ```js
    import { logEvent } from '../../utils/logEvent.js';

    /**
     * POST /api/ai/simulate-call
     * Generates a voice simulation script between AI and a participant.
     * Returns structured script for step-by-step frontend replay.
     */
    export const simulateCall = async (req, res, next) => {
      const { meetingId, participantName, participantEmail } = req.body;
      const userId = req.user.userId;

      try {
        // Log initiation event
        await logEvent({
          type: 'SIMULATION',
          message: `Initiating AI call to ${participantName} (${participantEmail})…`,
          status: 'pending',
          meetingId,
          userId,
          metadata: { participantName, participantEmail },
        });

        // Fetch meeting context
        const { PrismaClient } = await import('@prisma/client');
        const prisma = new PrismaClient();
        const meeting = await prisma.meeting.findUnique({
          where: { id: meetingId },
          select: { title: true, startTime: true, endTime: true },
        });

        if (!meeting) {
          const err = new Error('Meeting not found');
          err.statusCode = 404;
          throw err;
        }

        // Generate conversation script via AI
        const prompt = `Generate a realistic, professional phone call script between an AI meeting scheduler named "Exora" and ${participantName} regarding the meeting "${meeting.title}" scheduled from ${meeting.startTime.toISOString()} to ${meeting.endTime.toISOString()}. The AI is checking if ${participantName} can attend and discussing any scheduling concerns. Return ONLY valid JSON in this exact format: {"script": [{"speaker": "Exora", "line": "..."}, {"speaker": "${participantName}", "line": "..."}]} with 6-8 alternating turns. No markdown, no explanation.`;

        const aiService = await import('../../services/ai.service.js');
        const rawResult = await aiService.generateCompletion({
          taskType: 'CUSTOM',
          input: { prompt },
          metadata: { userId, meetingId },
        });

        // Parse script — handle both raw string and structured responses
        let script = [];
        try {
          const parsed = typeof rawResult === 'string' ? JSON.parse(rawResult) : rawResult;
          script = parsed.script || parsed.content?.script || [];
        } catch {
          script = [
            { speaker: 'Exora', line: `Hello ${participantName}, I'm calling about the meeting "${meeting.title}".` },
            { speaker: participantName, line: 'Yes, I received the invitation. Can you tell me more?' },
            { speaker: 'Exora', line: `The meeting is scheduled for ${meeting.startTime.toISOString()}. Does that work for you?` },
            { speaker: participantName, line: 'Let me check my calendar... Yes, that should be fine.' },
          ];
        }

        await logEvent({
          type: 'SIMULATION',
          message: `Call script generated for ${participantName} — ${script.length} turns`,
          status: 'success',
          meetingId,
          userId,
          metadata: { participantName, turnCount: script.length },
        });

        res.json({ success: true, data: { script, participantName, meetingId } });
      } catch (err) {
        await logEvent({
          type: 'ERROR',
          message: `Call simulation failed for ${participantName}: ${err.message}`,
          status: 'failed',
          meetingId,
          userId: req.user?.userId,
          metadata: { participantName, error: err.message },
        });
        next(err);
      }
    };
    ```
